def parseFunderType(funder):
    if funder[0]['person'] != None: 
        funderType = 'person'
    elif funder[0]['company'] != None: 
        funderType = 'company'
    elif funder[0]['financial_org'] != None: 
        funderType = 'financial_org'
    else:
        funderType = 'unattributed'
    return funderType
